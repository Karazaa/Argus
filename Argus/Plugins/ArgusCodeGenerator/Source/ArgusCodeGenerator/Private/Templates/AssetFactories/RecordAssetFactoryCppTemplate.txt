#include "#####Factory.h"
#include "ArgusEditorModule.h"

U#####Factory::U#####Factory(const FObjectInitializer& objectInitializer)
{
	SupportedClass = U#####::StaticClass();
	bEditAfterNew = true;
	bCreateNew = true;
}

UObject* U#####Factory::FactoryCreateNew(UClass* classDefinition, UObject* parentObject, FName name, EObjectFlags objectFlags, UObject* context, FFeedbackContext* warning)
{
	if (!ensure(classDefinition) || !ensure(classDefinition->IsChildOf(SupportedClass)))
	{
		return nullptr;
	}

	return NewObject<UObject>(parentObject, SupportedClass, name, objectFlags | RF_Transactional, context);
}

uint32 FAssetTypeActions_#####::GetCategories()
{
	return ArgusEditorModule::GetAssetTypeCategory();
}
